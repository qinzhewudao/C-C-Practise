/*一只青蛙一次可以跳上1级台阶，也可以跳上2级。求该青蛙跳上一个n级的台阶总共有多少种跳法。 */
class Solution {
public:
    int jumpFloor(int number) {
        if(number<=0) return 0;
        if(number<=1) return 1;
        int fn2 = 1;
        int fn1 = 1;//注意和fabonacci不同的地方
        while(number--)
        {
            fn1 += fn2;
            fn2 = fn1- fn2;
        }
        return fn2;
    }
};
